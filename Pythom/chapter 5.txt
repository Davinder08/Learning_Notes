# Python For loop
-------------------------------------------------------------------------------------------------------------
print("First Loop")
for v in range(4, 8):
 print(v,end = ' ')
 
print("\nSecond Loop")
for v in range(3, 9, 2):
 print(v, end = ' ')

print("\nThird Loop")
for v in range(5, 1, -1):
 print(v,end = ' ')

# Python break and continue
------------------------------------------------------------------------------------------------------------------
i = 5
while i > 1:
    if i == 3:
        i -= 1 
        continue
    print(i,end = ' ')
    i -= 1 
    if i == 2:
        break
